% =+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+
% Center Align Frame Titles and make bold 
% =+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+=+
% Redefine the frametitle template to 
% center-align all frame titles and make them bold
\setbeamerfont{frametitle}{series=\bfseries,size=\Large}
\setbeamertemplate{frametitle}{
    \vspace{0.2em}
    \begin{centering}
        \usebeamerfont{frametitle}\insertframetitle\par
    \end{centering}
    % \vspace{0.2em}
}
% ===========================



% Set font sizes globally (not 100% correct, but nearly works and is there. , unnoticable )
\setbeamerfont{frametitle}{size=\LARGE} % Frame titles
\setbeamerfont{framesubtitle}{size=\large} % Subtitles
\setbeamerfont{normal text}{size=\Large} % Regular text



% edit the footeer to makek page and date number:::
\setbeamertemplate{footline}{
  \leavevmode%
  \hbox{%
  \begin{beamercolorbox}[wd=.5\paperwidth,ht=2.25ex,dp=1ex,left]{date in head/foot}%
    \hspace*{2ex}\insertshortdate
  \end{beamercolorbox}%
  \begin{beamercolorbox}[wd=.5\paperwidth,ht=2.25ex,dp=1ex,right]{page number in head/foot}%
    \insertframenumber{} / \inserttotalframenumber\hspace*{2ex}
  \end{beamercolorbox}}%
  \vskip0pt%
}


% Adjust line spacing in the table of contents
